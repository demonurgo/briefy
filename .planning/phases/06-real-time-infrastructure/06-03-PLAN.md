```typescript
// Subscription Echo RT-02 — receber novos comentários em tempo real
useEffect(() => {
  if (!orgId || !window.Echo) return;

  const channel = window.Echo.private(`organization.${orgId}`);

  channel.listen('.demand.comment.created', (event: DemandCommentCreatedEvent) => {
    // Filtrar: apenas comentários desta demanda
    if (event.demandId !== demand.id) return;

    setComments(prev => {
      // Guard D-06: deduplicar — evitar duplicata se broadcast chegar após append otimista futuro
      if (prev.some(c => c.id === event.comment.id)) return prev;
      return [...prev, event.comment as Comment];
    });
  });

  return () => {
    // CRÍTICO: usar stopListening, NÃO window.Echo.leave()
    // leave() removeria o canal inteiro, matando a subscription RT-01 em Index.tsx
    channel.stopListening('.demand.comment.created');
  };
}, [orgId, demand.id]);
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-08 | Information Disclosure | Subscription `.demand.comment.created` em DemandDetailModal | mitigate | Canal `private-organization.{orgId}` autorizado via `routes/channels.php` — apenas membros da org recebem o broadcast; já coberto pelo canal compartilhado com RT-01 |
| T-06-09 | Tampering | Payload do comentário recebido via broadcast | accept | Payload construído 100% no servidor (DemandController::addComment); o cliente não pode injetar dados no broadcast; o frontend apenas lê e renderiza |
| T-06-10 | Spoofing | Listener `.demand.comment.created` sem filtro por demanda | mitigate | `if (event.demandId !== demand.id) return` filtra eventos de outras demandas da mesma org — sem este guard, todos os novos comentários da org apareceriam no modal aberto |
| T-06-11 | Denial of Service | Listener acumulado por re-render ou reabertura de modal | mitigate | Array de dependências `[orgId, demand.id]` garante que o useEffect reexecuta apenas quando necessário, cancelando o listener anterior antes de criar novo |
