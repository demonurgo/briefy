### 4. Create `app/Services/Ai/Telemetry/SpanEmitter.php`

Target ≥ 80 lines. AGPL header.

Design principle: **fully optional**. If OTEL classes missing OR endpoint unset, `emit()` runs the callable without instrumentation. Never throws due to telemetry.

```php
<?php
// (c) 2026 Briefy contributors — AGPL-3.0
namespace App\Services\Ai\Telemetry;

use Illuminate\Support\Facades\Log;
use Illuminate\Support\Facades\Redis;
use Throwable;

final class SpanEmitter
{
    private const PRICE_PER_MTOK = [
        'opus'   => ['input' => 15.00, 'output' => 75.00],
        'sonnet' => ['input' => 3.00,  'output' => 15.00],
        'haiku'  => ['input' => 1.00,  'output' => 5.00],
    ];

    private bool $enabled;

    public function __construct()
    {
        $this->enabled = (bool) config('services.otel.enabled', false);
    }

    /**
     * Wraps a callable in an OTEL span. Silent no-op if OTEL disabled.
     */
    public function emit(string $capability, int $organizationId, array $attributes, callable $callable): mixed
    {
        $startedAt = microtime(true);
        $tracer = null;
        $span = null;

        try {
            if ($this->enabled && class_exists(\OpenTelemetry\API\Globals::class)) {
                $tracer = \OpenTelemetry\API\Globals::tracerProvider()->getTracer('briefy-ai');
                $span = $tracer->spanBuilder("ai.{$capability}")->startSpan();
                $span->setAttribute('ai.capability', $capability);
                $span->setAttribute('organization_id', $organizationId);
                foreach ($attributes as $k => $v) {
                    if (is_scalar($v) || is_null($v)) {
                        $span->setAttribute($k, $v);
                    }
                }
            }
        } catch (Throwable $e) {
            // telemetry must never break the host call
            Log::debug('otel.span_start_failed', ['error' => class_basename($e)]);
            $span = null;
        }

        try {
            $result = $callable();
            return $result;
        } catch (Throwable $e) {
            if ($span) {
                try { $span->recordException($e); } catch (Throwable) {}
            }
            throw $e;
        } finally {
            $durationMs = (microtime(true) - $startedAt) * 1000;
            if ($span) {
                try {
                    $span->setAttribute('duration_ms', (int) $durationMs);
                    $span->end();
                } catch (Throwable) {}
            }
            // Always log locally (regardless of OTEL), cheap and useful.
            Log::info('ai.call', array_merge([
                'capability' => $capability,
                'organization_id' => $organizationId,
                'duration_ms' => (int) $durationMs,
            ], $attributes));
        }
    }

    /**
     * Record token usage + estimated cost after an AI call. Called separately
     * from emit() because the Anthropic SDK returns usage only on message_stop.
     */
    public function recordUsage(
        string $capability,
        int $organizationId,
        array $attributes,
        int $inputTokens,
        int $outputTokens,
        float $durationMs,
    ): void {
        $model = (string) ($attributes['model'] ?? '');
        $tier = str_contains($model, 'opus') ? 'opus'
              : (str_contains($model, 'sonnet') ? 'sonnet'
              : (str_contains($model, 'haiku') ? 'haiku' : null));

        $cost = null;
        if ($tier && isset(self::PRICE_PER_MTOK[$tier])) {
            $p = self::PRICE_PER_MTOK[$tier];
            $cost = ($inputTokens / 1_000_000) * $p['input']
                  + ($outputTokens / 1_000_000) * $p['output'];
        }

        Log::info('ai.usage', array_merge([
            'capability' => $capability,
            'organization_id' => $organizationId,
            'input_tokens' => $inputTokens,
            'output_tokens' => $outputTokens,
            'duration_ms' => (int) $durationMs,
            'cost_usd' => $cost ? round($cost, 5) : null,
        ], $attributes));

        // Redis counter for soft rate-limit surface.
        try {
            $date = now()->format('Y-m-d');
            Redis::incr("ai:calls:{$organizationId}:{$date}");
            Redis::expire("ai:calls:{$organizationId}:{$date}", 86400 * 7);
            if ($cost) {
                Redis::incrbyfloat("ai:cost_usd:{$organizationId}:{$date}", (float) $cost);
                Redis::expire("ai:cost_usd:{$organizationId}:{$date}", 86400 * 7);
            }
        } catch (Throwable $e) {
            Log::debug('ai.usage_redis_failed', ['error' => class_basename($e)]);
        }
    }
}
```

### 1. Create `app/Http/Middleware/AiUsageMeter.php`

```php
<?php
// (c) 2026 Briefy contributors — AGPL-3.0
namespace App\Http\Middleware;

use Closure;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Log;
use Illuminate\Support\Facades\Redis;

class AiUsageMeter
{
    private const SOFT_DAILY_LIMIT = 500;

    public function handle(Request $request, Closure $next): mixed
    {
        $startedAt = microtime(true);

        $response = $next($request);

        try {
            $user = $request->user();
            if (! $user || ! $user->organization_id) return $response;

            $orgId = $user->organization_id;
            $date = now()->format('Y-m-d');
            $key = "ai:http_calls:{$orgId}:{$date}";

            $count = (int) Redis::incr($key);
            if ($count === 1) {
                Redis::expire($key, 86400 * 7); // retain 7 days
            }

            // Soft alert — log only, never block in Phase 3.
            if ($count === self::SOFT_DAILY_LIMIT) {
                Log::warning('ai.soft_limit_reached', [
                    'organization_id' => $orgId,
                    'count' => $count,
                    'limit' => self::SOFT_DAILY_LIMIT,
                ]);
            }

            $durationMs = (microtime(true) - $startedAt) * 1000;
            Log::info('ai.http', [
                'route' => optional($request->route())->getName(),
                'organization_id' => $orgId,
                'user_id' => $user->id,
                'duration_ms' => (int) $durationMs,
                'status' => $response->getStatusCode(),
                'daily_count' => $count,
            ]);
        } catch (\Throwable $e) {
            Log::debug('ai_meter.failed', ['error' => class_basename($e)]);
        }

        return $response;
    }
}
```

### Pattern for streaming services (BriefStreamer, ChatStreamer)

Token accounting must happen on `message_stop` event, which carries cumulative usage. Update the foreach loop:

```php
// Inside the generator's try block:
$lastUsage = null;

foreach ($stream as $event) {
    // ...existing delta handling...

    if ($event->type === 'message_delta' || $event->type === 'message_stop') {
        // The cumulative usage appears in message_delta and/or message_stop events.
        if (isset($event->usage)) {
            $lastUsage = $event->usage;
        }
    }

    if ($event->type === 'message_stop') {
        // Record usage now (safe: usage field is present on stop).
        if ($lastUsage) {
            $this->emitter->recordUsage(
                'brief', // or 'chat' for ChatStreamer
                $demand->organization_id,
                ['demand_id' => $demand->id, 'model' => (string) config('services.anthropic.model_chat')],

                outputTokens: (int) ($lastUsage->outputTokens ?? $lastUsage->output_tokens ?? 0),
                durationMs: (microtime(true) - $streamStartedAt) * 1000,
            );
        }
        // ...existing persistence logic...
    }
}
```
