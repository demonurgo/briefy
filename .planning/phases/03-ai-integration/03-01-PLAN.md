```php
protected $appends = ['anthropic_api_key_mask'];       // ensure mask is always present in JSON

public function hasAnthropicKey(): bool
{
    return ! empty($this->anthropic_api_key_encrypted);
}

// Convenience accessor for server-side code (uses decrypted value thanks to 'encrypted' cast).
public function getAnthropicApiKeyAttribute(): ?string
{
    return $this->anthropic_api_key_encrypted;
}

// Mask accessor — ALWAYS returns null OR "sk-ant-...XXXX" (last 4 chars). Never the full key.
public function getAnthropicApiKeyMaskAttribute(): ?string
{
    $key = $this->anthropic_api_key_encrypted;
    if (empty($key)) { return null; }
    $tail = substr($key, -4);
    return 'sk-ant-...' . $tail;
}
```

```php
<?php
// (c) 2026 Briefy contributors — AGPL-3.0
namespace App\Services\Ai;

use Anthropic\Client;
use Anthropic\RequestOptions;
use App\Models\Organization;

final class AnthropicClientFactory
{
    /**
     * Resolve an Anthropic\Client scoped to a single organization's API key (BYOK).
     *
     * Aborts 402 (Payment Required) if the org has no key — callers that want graceful
     * degradation should call Organization::hasAnthropicKey() first and render the
     * "configure sua chave" UX instead of invoking this method.
     */
    public function forOrganization(Organization $org): Client
    {
        $key = $org->anthropic_api_key; // decrypted via Laravel 'encrypted' cast
        abort_if(empty($key), 402, 'No Anthropic API key configured for this organization.');

        return new Client(
            apiKey: $key,
            requestOptions: RequestOptions::with(maxRetries: 2, timeout: 120),
        );
    }

    /**
     * Dev/test fallback — uses config('services.anthropic.api_key_fallback') from .env.
     * Production code paths must use forOrganization().
     */
    public function forTesting(): Client
    {
        return new Client(
            apiKey: (string) config('services.anthropic.api_key_fallback'),
            requestOptions: RequestOptions::with(maxRetries: 2, timeout: 120),
        );
    }
}
```

### 2. Create `app/Http/Controllers/Settings/AiController.php`

`testKey()` performs TWO probes:
- Primary: minimal `messages->create` (1 token, cheap model) — confirms the key authenticates against the core API.
- Secondary: `GET https://api.anthropic.com/v1/agents?limit=0` with the `anthropic-beta: managed-agents-2026-04-01` header — confirms the account has Managed Agents beta enabled. When this returns 403/404, chat still works but MA-dependent features (D-35..D-40) must be disabled in the UI.

Response shape: `{ok: bool, chat_ok: bool, managed_agents_ok: bool, message: string, error_class?: string}`. The frontend (and downstream Clients/Index render) uses `managed_agents_ok` to decide whether "🤖 Conhecer este cliente com IA" is clickable.

```php
<?php
// (c) 2026 Briefy contributors — AGPL-3.0
namespace App\Http\Controllers\Settings;

use App\Http\Controllers\Controller;
use App\Http\Requests\TestAnthropicKeyRequest;
use Anthropic\Client;
use Anthropic\RequestOptions;
use Illuminate\Http\RedirectResponse;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Http;
use Inertia\Inertia;
use Inertia\Response;

class AiController extends Controller
{
    public function edit(): Response
    {
        $user = auth()->user();
        abort_unless(in_array($user->role, ['owner', 'admin']), 403);

        return Inertia::render('Settings/Ai', [
            'organization' => [
                'id' => $user->organization->id,
                'name' => $user->organization->name,
                'has_anthropic_key' => $user->organization->hasAnthropicKey(),
                'anthropic_api_key_mask' => $user->organization->anthropic_api_key_mask,
            ],
        ]);
    }

    public function update(Request $request): RedirectResponse
    {
        $user = auth()->user();
        abort_unless(in_array($user->role, ['owner', 'admin']), 403);

        $request->validate([
            'anthropic_api_key' => ['nullable', 'string', 'starts_with:sk-ant-', 'min:30', 'max:200'],
        ]);

        $user->organization->update([
            'anthropic_api_key_encrypted' => $request->input('anthropic_api_key') ?: null,
        ]);

        return back()->with('success', __('app.ai_key_saved'));
    }

    public function testKey(TestAnthropicKeyRequest $request): \Illuminate\Http\JsonResponse
    {
        $user = auth()->user();
        abort_unless(in_array($user->role, ['owner', 'admin']), 403);

        $key = (string) $request->input('anthropic_api_key');
        $chatOk = false;
        $maOk = false;
        $errorClass = null;

        // Primary probe: minimal chat call. Confirms key authenticates against core API.
        try {
            $client = new Client(
                apiKey: $key,
                requestOptions: RequestOptions::with(maxRetries: 0, timeout: 15),
            );
            $client->messages->create(
                maxTokens: 1,
                messages: [['role' => 'user', 'content' => 'ping']],
                model: (string) config('services.anthropic.model_cheap'),
            );
            $chatOk = true;
        } catch (\Throwable $e) {
            $errorClass = class_basename($e);
        }

        // Secondary probe: Managed Agents beta access. Cheap (no tokens consumed, list call).
        // Surfaces to UI so D-35..D-40 features can be disabled when MA is not available.
        if ($chatOk) {
            try {
                $resp = Http::withHeaders([
                    'authorization'     => "Bearer {$key}",
                    'anthropic-beta'    => (string) config('services.anthropic.beta_ma'),
                    'anthropic-version' => '2023-06-01',
                ])->timeout(10)->get('https://api.anthropic.com/v1/agents', ['limit' => 0]);

                // Treat 200 and 204 as success. 403/404 = MA beta not enabled on this account.
                $maOk = in_array($resp->status(), [200, 204], true);
            } catch (\Throwable $e) {
                // Network hiccups etc. — conservatively report MA as unavailable without failing the whole test.
                $maOk = false;
            }
        }

        $ok = $chatOk; // overall success == chat works. MA is informational, not a hard requirement.

        return response()->json([
            'ok' => $ok,
            'chat_ok' => $chatOk,
            'managed_agents_ok' => $maOk,
            'message' => $ok
                ? ($maOk ? __('app.ai_key_valid') : __('app.ai_key_valid_no_ma'))
                : __('app.ai_key_invalid'),
            'error_class' => $errorClass,
        ], $ok ? 200 : 422);
    }
}
```

```tsx
// (c) 2026 Briefy contributors — AGPL-3.0
import { useState } from 'react';
import { Head, useForm, router } from '@inertiajs/react';
import { useTranslation } from 'react-i18next';
import { Check, X, CircleAlert, Loader2 } from 'lucide-react';
import AppLayout from '@/Layouts/AppLayout';
import InputError from '@/Components/InputError';

interface Props {
  organization: { id: number; name: string; has_anthropic_key: boolean; anthropic_api_key_mask: string|null; };
}

interface TestResult {
  ok: boolean;
  chat_ok: boolean;
  managed_agents_ok: boolean;
  message: string;
}

export default function SettingsAi({ organization }: Props) {
  const { t } = useTranslation();
  const [testStatus, setTestStatus] = useState<null | 'testing' | 'valid' | 'invalid'>(null);
  const [testResult, setTestResult] = useState<TestResult | null>(null);
  const form = useForm({ anthropic_api_key: '' });

  const submit = (e: React.FormEvent) => {
    e.preventDefault();
    form.patch(route('settings.ai.update'), { preserveScroll: true, onSuccess: () => form.reset() });
  };

  const handleTest = async () => {
    if (!form.data.anthropic_api_key) return;
    setTestStatus('testing');
    try {
      const res = await fetch(route('settings.ai.test'), {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'X-CSRF-TOKEN': document.querySelector<HTMLMetaElement>('meta[name="csrf-token"]')?.content ?? '' },
        body: JSON.stringify({ anthropic_api_key: form.data.anthropic_api_key }),
      });
      const data: TestResult = await res.json();
      setTestResult(data);
      setTestStatus(data.ok ? 'valid' : 'invalid');
    } catch {
      setTestStatus('invalid');
      setTestResult({ ok: false, chat_ok: false, managed_agents_ok: false, message: t('settings.ai.testFailed') });
    }
  };

  return (
    <AppLayout>
      <Head title={t('settings.ai.title')} />
      <div className="max-w-2xl mx-auto p-6">
        <h1 className="text-xl font-semibold mb-1">{t('settings.ai.title')}</h1>
        <p className="text-sm text-[#6b7280] mb-6">{t('settings.ai.description')}</p>

        {/* Status badge */}
        <div className="mb-6 flex items-center gap-2 rounded-[8px] border border-[#e5e7eb] dark:border-[#1f2937] bg-white dark:bg-[#111827] px-4 py-3">
          {organization.has_anthropic_key ? (
            <><Check size={16} className="text-green-500" /><span className="text-sm">{t('settings.ai.status.configured')}: <code className="font-mono">{organization.anthropic_api_key_mask}</code></span></>
          ) : (
            <><CircleAlert size={16} className="text-[#f59e0b]" /><span className="text-sm">{t('settings.ai.status.missing')}</span></>
          )}
        </div>

        <form onSubmit={submit} className="space-y-4">
          <div>
            <label className="mb-1.5 block text-sm font-medium">{t('settings.ai.keyLabel')}</label>
            <input type="password" autoComplete="off"
              value={form.data.anthropic_api_key}
              onChange={e => { form.setData('anthropic_api_key', e.target.value); setTestStatus(null); setTestResult(null); }}
              placeholder="sk-ant-api03-..."
              className="w-full rounded-[8px] border border-[#e5e7eb] bg-white px-3.5 py-2.5 text-sm font-mono focus:border-[#7c3aed] focus:outline-none focus:ring-2 focus:ring-[#7c3aed]/20 dark:border-[#1f2937] dark:bg-[#111827] dark:text-[#f9fafb]" />
            <InputError message={form.errors.anthropic_api_key} className="mt-1.5" />
            <p className="text-xs text-[#9ca3af] mt-1">{t('settings.ai.keyHint')}</p>
          </div>

          <div className="flex items-center gap-3">
            <button type="button" onClick={handleTest} disabled={!form.data.anthropic_api_key || testStatus === 'testing'}
              className="rounded-[8px] border border-[#e5e7eb] dark:border-[#1f2937] px-3 py-1.5 text-sm font-medium text-[#6b7280] hover:border-[#7c3aed] hover:text-[#7c3aed] disabled:opacity-50">
              {testStatus === 'testing' ? <Loader2 size={14} className="animate-spin inline mr-1" /> : null}
              {t('settings.ai.testKey')}
            </button>
            {testStatus === 'valid' && testResult && (
              <span className="text-sm text-green-600 flex items-center gap-1">
                <Check size={14} /> {testResult.message}
              </span>
            )}
            {testStatus === 'invalid' && testResult && (
              <span className="text-sm text-red-500 flex items-center gap-1">
                <X size={14} /> {testResult.message}
              </span>
            )}
          </div>

          {/* MA availability notice — shown only when chat works but MA does not. */}
          {testResult?.chat_ok && !testResult.managed_agents_ok && (
            <div className="rounded-[8px] border border-[#f59e0b]/30 bg-[#f59e0b]/10 p-3 text-xs text-[#b45309] dark:text-[#f59e0b]">
              {t('settings.ai.maUnavailableBanner')}
            </div>
          )}

          <button type="submit" disabled={form.processing}
            className="rounded-[8px] bg-[#7c3aed] hover:bg-[#6d28d9] px-4 py-2 text-sm font-medium text-white disabled:opacity-60">
            {form.processing ? t('common.saving') : t('common.save')}
          </button>
        </form>
      </div>
    </AppLayout>
  );
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→API (settings.ai.update) | User pastes Anthropic API key — untrusted input that, once stored, authorizes Claude API calls on the org's bill |
| API→DB | Encrypted column `anthropic_api_key_encrypted` — crypto depends on APP_KEY integrity |
| API→Anthropic | testKey() proxies user-pasted key to api.anthropic.com — small-blast-radius but immediate cost exposure |
| Inertia→Browser | Shared props must mask the key; a regression that leaks `anthropic_api_key_encrypted` exposes every org |
